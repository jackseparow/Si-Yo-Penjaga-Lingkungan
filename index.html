function calculateEco(block) { if (!block) return null; if (block.type === 'mixer_eco') { const pBlock = block.getInputTargetBlock('P'); // Slot Polutan const fBlock = block.getInputTargetBlock('F'); // Slot Solusi let basePH = 8.1; // pH Laut Alami let impact = 0; // Efek Polusi (Biasanya negatif) let recovery = 0; // Efek Solusi (Harus positif) // 1. Ambil nilai dari Polutan if (pBlock) { // Misalnya: (pH Polutan 4.0 - 7.0) = -3.0 impact = parseFloat(pBlock.getFieldValue('TYPE')) - 7.0; } // 2. Ambil nilai dari Solusi (BAGIAN PENTING!) if (fBlock) { // Ambil nilai kekuatan netralisir dari dropdown solusi recovery = parseFloat(fBlock.getFieldValue('TYPE')); } // 3. Hitung Total Akhir let finalPH = basePH + impact + recovery; // Jangan biarkan laut terlalu basa (maksimal 8.5) if (finalPH > 8.5) finalPH = 8.5; return { ph: finalPH, hasPoll: !!pBlock, hasFix: !!fBlock, fixIcon: fBlock ? fBlock.getFieldValue('TYPE') : null }; } return null; }
